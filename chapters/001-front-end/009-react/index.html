import React, { useState } from 'react';

// custom hook
export function useCounter (initialCount) {
  const [count, setCount] = useState(initialCount);
  return {
    count, // same as count: count,
    decrement: (decrementBy = 1) => { // default parameter
      setCount(count - decrementBy);
    },
    increment: (incrementBy = 1) => { // default parameter
      setCount(count + incrementBy);
    }
  }
} 

export const Counter = () => {
  const { count, decrement, increment } = useCounter(0);

  return (
    <div>
      {/* decrement will default to 1 */}                                   
      <button onClick={() => { decrement() }}>-</button>
      {count}
      <button onClick={() => { increment(2) }}>+</button>
    </div>
  );
};
const Toggle = ({ content }) => {
  const [isVisible, setIsVisible] = useState(false);

  return (
    <div>
      <button onClick={() => { setIsVisible(!isVisible); }}>
        Toggle!
      </button>
      {/* conditional rendering */}
      {isVisible ? <div>{content}</div> : null}
    </div>
  );
};
