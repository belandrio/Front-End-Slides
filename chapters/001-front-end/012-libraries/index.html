Applying classNames 
import React from 'react';

function Button ({ size = 'large', variant = 'primary', children, disabled = false }) {
  let buttonClassName = `btn btn-${variant}`; // btn btn-primary
  if(disabled) {
    buttonClassName += ' btn-disabled';
  }

  switch(size) {
    case 'small':
      buttonClassName += ' btn-xs';
      break;
    case 'large':
      buttonClassName += ' btn-2x';
      break;
    case 'x-large':
      buttonClassName += ' btn-3x';
      break;
    default:
  }

  return <button className={buttonClassName}>{children}</button>
}

ClassNames

import React from 'react';
import classNames from 'classnames';

function Button ({ size = 'large', variant = 'primary', children, disabled = false }) {
  const buttonClassName = classNames(`btn btn-${variant}`, {
    'btn-disabled': disabled,
    'btn-xs': size === 'small',
    'btn-2x': size === 'large',
    'btn-3x': size === 'x-large'
  })
  
  return <button className={buttonClassName}>{children}</button>
}
